# 🚀 Step 3 — Create Model
📄 app1/models.py
```python
from django.db import models

class Student(models.Model):
    name = models.CharField(max_length=100)
    age = models.IntegerField()
    city = models.CharField(max_length=100)

    def __str__(self):
        return self.name
```

# 🚀 Step 5 — Create Views
📄 app1/views.py

```python
from django.shortcuts import render, redirect
from .models import Student

# CREATE
def add_student(request):

    if request.method == 'POST':
        name = request.POST.get('name')
        age = request.POST.get('age')
        city = request.POST.get('city')

        Student.objects.create(name=name, age=age, city=city)
        return redirect('list')

    return render(request, 'add.html')


# READ
def student_list(request):
    students = Student.objects.all()
    return render(request, 'list.html', {'students': students})


# UPDATE (NO URL PARAMS)
def update_student(request):

    if request.method == 'POST':
        id = request.POST.get('id')
        student = Student.objects.get(id=id)

        student.name = request.POST.get('name')
        student.age = request.POST.get('age')
        student.city = request.POST.get('city')
        student.save()

        return redirect('list')

    return redirect('list')

# DELETE
def delete_student(request):

    if request.method == 'POST':
        id = request.POST.get('id')
        student = Student.objects.get(id=id)
        student.delete()

    return redirect('list')
```

## 📄 templates/list.html (READ + UPDATE + DELETE)

```html
<h2>Student List</h2>

<a href="{% url 'add' %}">Add Student</a>

<table border="1">
<tr>
    <th>Name</th>
    <th>Age</th>
    <th>City</th>
    <th>Actions</th>
</tr>

{% for s in students %}
<tr>
<form method="POST">
{% csrf_token %}

    <td>
        <input type="text" name="name" value="{{ s.name }}">
    </td>

    <td>
        <input type="number" name="age" value="{{ s.age }}">
    </td>

    <td>
        <input type="text" name="city" value="{{ s.city }}">
    </td>

    <td>
        <input type="hidden" name="id" value="{{ s.id }}">

        <button formaction="{% url 'update' %}">Update</button>
        <button formaction="{% url 'delete' %}">Delete</button>
    </td>

</form>
</tr>
{% endfor %}

</table>
```

# 🚀 Step 8 — Configure URLs
📄 CRUDProject/urls.py
```python
from django.contrib import admin
from django.urls import path
from app1 import views

urlpatterns = [
    path('admin/', admin.site.urls),

    path('', views.student_list, name='list'),
    path('add/', views.add_student, name='add'),
    path('update/', views.update_student, name='update'),
    path('delete/', views.delete_student, name='delete'),
]
```
