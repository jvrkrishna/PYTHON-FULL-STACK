# Step 3 — Create Model
📄 myapp/models.py
```python
from django.db import models

class Post(models.Model):

    title = models.CharField(max_length=100)
    content = models.TextField()
    author = models.CharField(max_length=50, default="Anonymous")
    created_at = models.DateTimeField(auto_now_add=True)

    def __str__(self):
        return self.title
```

# Step 7 — Create Views (Model → View)
📄 myapp/views.py
```python
from django.shortcuts import render
from .models import Post

# Show all posts
def myapp_view(request):
    posts = Post.objects.all()
    return render(request, 'myapp/myapp_home.html', {'posts': posts})


# Filter posts by author
def rk_posts_view(request):
    posts = Post.objects.filter(author="RK")
    return render(request, 'myapp/myapp_home.html', {'posts': posts})
```

---
# Step 8 — Configure URLs
📄 myapp/urls.py
```python
from django.urls import path
from . import views

    path('', views.myapp_view, name='all_posts'),
    path('rk/', views.rk_posts_view, name='rk_posts'),
]
```

---
# Step 9 — Create Template
📄 myapp/templates/myapp/myapp_home.html
```html
<!DOCTYPE html>
<html>
<head>
    <title>Blog Posts</title>
</head>
<body>

<h1>Blog Posts</h1>

{% for i in posts %}
    <h3>{{ i.title }}</h3>
    <i>{{ i.created_at }}</i><br>
    <b>By: {{ i.author }}</b>
    <p>{{ i.content }}</p>
    <hr>

{% empty %}
    <p>No posts available.</p>
{% endfor %}

</body>
</html>
```
