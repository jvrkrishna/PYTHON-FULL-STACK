# Step 4 — Create Model
📄 myapp/models.py
```python
from django.db import models

class Post(models.Model):

    title = models.CharField(max_length=100)
    content = models.TextField()
    author = models.CharField(max_length=50, default="Anonymous")
    created_at = models.DateTimeField(auto_now_add=True)

    def __str__(self):
        return self.title
```

# Step 7 — Create Views (CRUD)
📄 myapp/views.py
```python
from django.shortcuts import render, redirect, get_object_or_404
from .models import Post
from .forms import PostForm

# READ
def post_list(request):
    posts = Post.objects.all()
    return render(request, 'myapp/post_list.html', {'posts': posts})

# CREATE
def post_create(request):
    if request.method == "POST":
        form = PostForm(request.POST)
        if form.is_valid():
            form.save()
            return redirect('post_list')
    else:
        form = PostForm()

    return render(request, 'myapp/post_form.html', {'form': form})

# UPDATE
def post_update(request, pk):
    post = get_object_or_404(Post, pk=pk)

    if request.method == "POST":
        form = PostForm(request.POST, instance=post)
        if form.is_valid():
            form.save()
            return redirect('post_list')
    else:
        form = PostForm(instance=post)

    return render(request, 'myapp/post_form.html', {'form': form})

# DELETE
def post_delete(request, pk):
    post = get_object_or_404(Post, pk=pk)

    if request.method == "POST":
        post.delete()
        return redirect('post_list')

    return render(request, 'myapp/post_confirm_delete.html', {'post': post})
```

# Step 8 — Configure URLs
📄 myapp/urls.py
```python
from django.urls import path
from . import views

urlpatterns = [
    path('', views.post_list, name='post_list'),
    path('create/', views.post_create, name='post_create'),
    path('<int:pk>/edit/', views.post_update, name='post_update'),
    path('<int:pk>/delete/', views.post_delete, name='post_delete'),
]
```

---
## post_list.html
```html
<h1>All Posts</h1>

<a href="{% url 'post_create' %}">Add New Post</a>
<hr>

{% for post in posts %}
    <h3>{{ post.title }}</h3>
    <p>{{ post.content }}</p>
    <i>{{ post.author }} | {{ post.created_at }}</i><br>

    <a href="{% url 'post_update' post.pk %}">Edit</a> |
    <a href="{% url 'post_delete' post.pk %}">Delete</a>
    <hr>

{% empty %}
    <p>No posts available.</p>
{% endfor %}
```

---
## post_form.html
```html
<h1>{% if form.instance.pk %}Edit{% else %}Create{% endif %} Post</h1>

</form>
```
